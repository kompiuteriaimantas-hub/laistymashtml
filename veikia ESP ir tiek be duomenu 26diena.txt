float readMoisturePercent() {
  int raw = analogRead(MOISTURE_PIN);
  float percent = 100.0 * (float)(dry - raw) / (float)(dry - wet);
  if (percent < 0) percent = 0;
  if (percent > 100) percent = 100;
  return percent;
}

bool ensureWiFi() {
  if (WiFi.status() == WL_CONNECTED) {
    wifiOnline = true;
    return true;
  }

  unsigned long now = millis();

  if (now - wifiRetryWindowStart > WIFI_RETRY_WINDOW) {
    wifiRetryWindowStart = now;
    wifiRetryCount = 0;
  }

  if (wifiRetryCount >= WIFI_RETRY_LIMIT) {
    wifiOnline = false;
    return false;
  }

  wifiRetryCount++;

  WiFi.begin(ssid, password);
  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - start < 8000) {
    delay(200);
  }

  wifiOnline = (WiFi.status() == WL_CONNECTED);
  return wifiOnline;
}

void updateDisplay(const char* statusText, float moisture, float temp, float pressure) {

  display.setTextColor(SSD1306_WHITE);

  display.setTextSize(2);
  display.setCursor(0, 0);
  display.println(statusText);

  display.setTextSize(1);
  display.setCursor(0, 28);
  display.print("Dregme: ");
  display.print(moisture, 1);
  display.println("%");

  display.setCursor(0, 45);
  display.print(temp, 1);
  display.print("C  ");
  display.print((int)pressure);
  display.println("hPa");

  display.display();
}

void sendDataUsage() {
  if (!ensureWiFi()) return;

  HTTPClient http;
  WiFiClientSecure client;
  client.setInsecure();

  float kb = dataUsedBytes / 1024.0;

  String url = String(SUPABASE_DATA_URL) + "?id=eq." + ESP_ID;

  if (http.begin(client, url)) {
    http.addHeader("Content-Type", "application/json");
    http.addHeader("apikey", API_KEY);
    http.addHeader("Authorization", String("Bearer ") + API_KEY);
    http.addHeader("Prefer", "resolution=merge-duplicates");

    String payload = "{";
    payload += "\"id\":\"" + String(ESP_ID) + "\",";
    payload += "\"data_used_kb\":" + String(kb, 3);
    payload += "}";

    // PATCH
    int code = http.PATCH(payload);

    // DEBUG
    Serial.println("----- DATA USAGE DEBUG -----");
    Serial.println("Sending KB: " + String(kb, 3));
    Serial.println("Payload bytes: " + String(payload.length()));
    Serial.println("HTTP code: " + String(code));

    // RESPONSE (VIENAS KARTAS)
    String resp = http.getString();
    Serial.println("Response: " + resp);

    // SKAIČIUOJAM USAGE
    addDataUsage(payload.length());
    addDataUsage(resp.length());

    Serial.println("-----------------------------");

    http.end();
  }
}

  sendStatus();
}

      }

      if (doReset || doUnblock || relayOnCmd || relayOffCmd || relayToggleCmd) {
        clearCommandsOnServer();
      }
    }

    http.end();
  }


void autoWaterCycle() {
  // Jei buvo UNLOCK komanda — iškart ją sunaikinam Supabase
 if (!lockdown) {
    Serial.println("[AUTO] Clear unblock on start");
    clearCommandsOnServer();   // ← ČIA MAGIJA
  }

  if (lockdown) return;
  if (isWatering) return;

  Serial.println("[AUTO] Dregme maza. Pradedu laistyma.");

  float before = 0;
  float after = 0;

  for (int attempt = 1; attempt <= 3; attempt++) {

    Serial.printf("[WATER] Bandymas %d\n", attempt);

    refreshSensors();
    before = lastMoisture;

    relayOn();
    updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    delay(10000);
    relayOff();

    for (int i = 0; i < 3; i++) {
      delay(10000);
      updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    }

    refreshSensors();
    after = lastMoisture;

    if (after >= before + 3.0) {
      updateDisplay("OK", lastMoisture, lastTemp, lastPressure);
      Serial.println("[AUTO] Dregme pakilo. Baigiam.");
      return;
    }
  }

  Serial.println("[LOCKDOWN] Dregme nekyla. Pereinam į LOCKDOWN.");

  lockdown = true;
  lockdownReported = false;

  sendError("dregme nekyla");
  sendStatus();
  lockdownReported = true;

  updateDisplay("LOCK DOWN", lastMoisture, lastTemp, lastPressure);
}




void setup() {
  Serial.begin(115200);
  delay(200);

  WiFi.mode(WIFI_STA);
  WiFi.setSleep(false);
  WiFi.begin(ssid, password);

  Serial.println("Connecting WiFi...");
  while (WiFi.status() != WL_CONNECTED) {
    delay(200);
    Serial.print(".");
  }
  Serial.println("\nWiFi OK");

  // NTP po WiFi
  configTime(0, 0, "pool.ntp.org", "time.nist.gov");
  Serial.println("Syncing time...");
  delay(2000);

  time_t now = time(nullptr);
  while (now < 100000) {
    delay(500);
    Serial.print(".");
    now = time(nullptr);
  }
  Serial.println("\nTime synced!");

  // OLED tik po NTP
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  display.clearDisplay();
  display.display();

  // BMP
  bmp.begin(0x77);

  refreshSensors();
  updateDisplay("OK", lastMoisture, lastTemp, lastPressure);

  pinMode(RELAY_PIN, OUTPUT);
  relayOff();

  randomSeed(analogRead(MOISTURE_PIN));

 // sendStatus();
//sendDataUsage();
//checkCommands();
lastStatusSend = millis();
}


void loop() {
  unsigned long now = millis();

  

  static unsigned long lastCheck = 0;
  if (now - lastCheck > 5000) {
    lastCheck = now;

    refreshSensors();

    if (isWatering) {
      updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    }
    else if (lockdown) {
      updateDisplay("LOCK DOWN", lastMoisture, lastTemp, lastPressure);
    }
    else if (lastMoisture < 30.0) {
      autoWaterCycle();
    }
    else {
      updateDisplay("OK", lastMoisture, lastTemp, lastPressure);
    }
  }

  if ((now - lastStatusSend > STATUS_INTERVAL || lastStatusSend == 0)) {
    if (true) {
      sendStatus();
      sendDataUsage();
      checkCommands();
      lastStatusSend = now;
    }
  }
}
