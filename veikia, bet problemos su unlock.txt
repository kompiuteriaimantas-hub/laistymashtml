void syncTime() {

float readMoisturePercent() {
  int raw = analogRead(MOISTURE_PIN);
  float percent = 100.0 * (float)(dry - raw) / (float)(dry - wet);
  if (percent < 0) percent = 0;
  if (percent > 100) percent = 100;
  return percent;
}

bool ensureWiFi() {
  if (WiFi.status() == WL_CONNECTED) {
    wifiOnline = true;
    return true;
  }

  WiFi.disconnect();
  WiFi.begin(ssid, password);

  unsigned long start = millis();
  while (WiFi.status() != WL_CONNECTED && millis() - start < 8000) {
    delay(200);
  }

  wifiOnline = (WiFi.status() == WL_CONNECTED);
  return wifiOnline;
}

}


void autoWaterCycle() {
  if (lockdown) return;
  if (isWatering) return;

  Serial.println("[AUTO] Dregme maza. Pradedu laistyma.");

  float before = 0;
  float after = 0;

  for (int attempt = 1; attempt <= 3; attempt++) {

    Serial.printf("[WATER] Bandymas %d\n", attempt);

    refreshSensors();
    before = lastMoisture;

    relayOn();
    updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    delay(10000);
    relayOff();

    for (int i = 0; i < 3; i++) {
      delay(10000);
      updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    }

    refreshSensors();
    after = lastMoisture;

    if (after >= before + 3.0) {
      updateDisplay("OK", lastMoisture, lastTemp, lastPressure);
      Serial.println("[AUTO] Dregme pakilo. Baigiam.");
      return;
    }
  }

  
Serial.println("[LOCKDOWN] Dregme nekyla. Pereinam į LOCKDOWN.");

lockdown = true;
lockdownReported = false;
unblockConsumed = false;   // ✅ LEIDŽIA kitą unlock


  sendError("dregme nekyla");
  sendStatus();
  lockdownReported = true;

  updateDisplay("LOCK DOWN", lastMoisture, lastTemp, lastPressure);
}

void setup() {
  Serial.begin(115200);
  delay(200);

  WiFi.mode(WIFI_STA);
  WiFi.setSleep(false);
  WiFi.begin(ssid, password);

  Serial.println("Connecting WiFi...");
  while (WiFi.status() != WL_CONNECTED) {
    delay(200);
    Serial.print(".");
  }
  Serial.println("\nWiFi OK");

  // NTP po WiFi
  configTime(0, 0, "pool.ntp.org", "time.nist.gov");
  Serial.println("Syncing time...");
  delay(2000);

  time_t now = time(nullptr);
  while (now < 100000) {
    delay(500);
    Serial.print(".");
    now = time(nullptr);
  }
  Serial.println("\nTime synced!");

  // OLED tik po NTP
  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  display.clearDisplay();
  display.display();

  // BMP
  bmp.begin(0x77);

  refreshSensors();
  updateDisplay("OK", lastMoisture, lastTemp, lastPressure);

  pinMode(RELAY_PIN, OUTPUT);
  relayOff();

  randomSeed(analogRead(MOISTURE_PIN));

  sendStatus();
sendDataUsage();
checkCommands();
lastStatusSend = millis();
}


void loop() {
  unsigned long now = millis();

  // Sensorių atnaujinimas kas 5 sekundes
  static unsigned long lastCheck = 0;
  if (now - lastCheck > 5000) {
    lastCheck = now;

    refreshSensors();

    // Manual relay komandos
    if (manualRelayRequest == 1) {
      relayOn();
      manualRelayRequest = 0;
    }
    else if (manualRelayRequest == 2) {
      relayOff();
      manualRelayRequest = 0;
    }
    else if (manualRelayRequest == 3) {
      if (isWatering) relayOff();
      else relayOn();
      manualRelayRequest = 0;
    }

    // Ekranas
    if (isWatering) {
      updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    }
    else if (lockdown) {
      updateDisplay("LOCK DOWN", lastMoisture, lastTemp, lastPressure);
    }
    else if (lastMoisture < 30.0) {
      autoWaterCycle();
    }
    else {
      updateDisplay("OK", lastMoisture, lastTemp, lastPressure);
    }
  }

  // Statuso siuntimas
  if ((now - lastStatusSend > STATUS_INTERVAL || lastStatusSend == 0)) {

    if (ensureWiFi()) {
      sendStatus();
      sendDataUsage();
      checkCommands();
    }

    lastStatusSend = now;
  }

  // PATCH vykdomas tik čia — SAUGIAI
  if (needClearCommands) {
    clearCommandsOnServer();
    needClearCommands = false;
  }
}
