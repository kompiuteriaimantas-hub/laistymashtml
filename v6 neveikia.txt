#include <ESP8266WiFi.h>
#include <ESP8266HTTPClient.h>
#include <WiFiClientSecure.h>
#include <Wire.h>
#include <Adafruit_BMP280.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// ---- PINAI ----
#define MOISTURE_PIN A0
#define RELAY_PIN    D5

// ---- OLED ----
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// ---- BMP280 ----
Adafruit_BMP280 bmp;

// ---- WiFi ----
const char* ssid     = "";
const char* password = "";

// ---- Supabase ----
const char* SUPABASE_STATUS_URL    = "https://wbueugwhngtgtiufasvm.supabase.co/functions/v1/status";
const char* SUPABASE_ERROR_URL     = "https://wbueugwhngtgtiufasvm.supabase.co/functions/v1/error";
const char* SUPABASE_DATA_URL      = "https://wbueugwhngtgtiufasvm.supabase.co/rest/v1/data_usage";
const char* SUPABASE_COMMANDS_URL  = "https://wbueugwhngtgtiufasvm.supabase.co/rest/v1/commands";

const char* ESP_ID = "esp1";

const char* API_KEY = "TAVO_KEY";

// ---- Moisture ----
const int dry = 650;
const int wet = 336;

// ---- Time ----
unsigned long lastStatusSend = 0;
unsigned long lastDataSend = 0;       // ✅ FIX
unsigned long lastCmdCheck = 0;       // ✅ FIX
const unsigned long STATUS_INTERVAL = 15000UL;  // ✅ FIX (buvo 5000)

bool lockdown = false;
bool lockdownReported = false;
bool wifiOnline = false;

// ---- Sensors ----
float lastMoisture = NAN;
float lastTemp = NAN;
float lastPressure = NAN;

// ---- Flags ----
bool isWatering = false;
unsigned long dataUsedBytes = 0;

void addDataUsage(size_t payloadLen) {
  dataUsedBytes += payloadLen + 300;
}

// ---- SENSOR ----
float readMoisturePercent() {
  int raw = analogRead(MOISTURE_PIN);
  float percent = 100.0 * (dry - raw) / (float)(dry - wet);
  if (percent < 0) percent = 0;
  if (percent > 100) percent = 100;
  return percent;
}

// ---- WIFI ----
bool ensureWiFi() {
  if (WiFi.status() == WL_CONNECTED) return true;

  WiFi.begin(ssid, password);
  unsigned long start = millis();

  while (WiFi.status() != WL_CONNECTED && millis() - start < 8000) {
    delay(200);
  }

  return WiFi.status() == WL_CONNECTED;
}

// ---- DISPLAY ----
void updateDisplay(const char* statusText, float moisture, float temp, float pressure) {
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);

  display.setTextSize(2);
  display.setCursor(0, 0);
  display.println(statusText);

  display.setTextSize(1);
  display.setCursor(0, 28);
  display.print("Dregme: ");
  display.println(moisture, 1);

  display.setCursor(0, 45);
  display.print(temp, 1);
  display.print("C ");
  display.print((int)pressure);
  display.println("hPa");

  display.display();
}

// ---- SENSOR REFRESH ----
void refreshSensors() {
  lastMoisture = readMoisturePercent();
  lastTemp = bmp.readTemperature();
  lastPressure = bmp.readPressure() / 100.0F;
}

// ---- RELAY ----
void relayOn() {
  digitalWrite(RELAY_PIN, HIGH);
  isWatering = true;
}

void relayOff() {
  digitalWrite(RELAY_PIN, LOW);
  isWatering = false;
}

// =======================
// ✅ HTTP FIX
// =======================

void sendStatus() {
  if (!ensureWiFi()) return;

  refreshSensors();

  HTTPClient http;
  WiFiClientSecure client;
  client.setInsecure();

  delay(50); // ✅ FIX

  if (http.begin(client, SUPABASE_STATUS_URL)) {

    http.addHeader("Content-Type", "application/json");
    http.addHeader("apikey", API_KEY);
    http.addHeader("Authorization", String("Bearer ") + API_KEY);

    String payload = "{";
    payload += "\"id\":\"" + String(ESP_ID) + "\",";
    payload += "\"moisture\":" + String(lastMoisture, 1) + ",";
    payload += "\"temp\":" + String(lastTemp, 2) + ",";
    payload += "\"pressure\":" + String(lastPressure, 2) + ",";
    payload += "\"lockdown\":" + String(lockdown ? "true" : "false");
    payload += "}";

    http.POST(payload);
    http.end();
    delay(50); // ✅ FIX
  }
}

void sendDataUsage() {
  if (!ensureWiFi()) return;

  HTTPClient http;
  WiFiClientSecure client;
  client.setInsecure();

  delay(50); // ✅ FIX

  String url = String(SUPABASE_DATA_URL) + "?id=eq." + ESP_ID;

  if (http.begin(client, url)) {

    http.addHeader("Content-Type", "application/json");
    http.addHeader("apikey", API_KEY);
    http.addHeader("Authorization", String("Bearer ") + API_KEY);

    float mb = dataUsedBytes / 1024.0 / 1024.0;

    String payload = "{";
    payload += "\"id\":\"" + String(ESP_ID) + "\",";
    payload += "\"data_used_mb\":" + String(mb, 3);
    payload += "}";

    http.POST(payload);
    http.end();
    delay(50); // ✅ FIX
  }
}

// =======================
// AUTO WATER – PALIKTA ✅
// =======================

void autoWaterCycle() {
  if (lockdown || isWatering) return;

  float before = readMoisturePercent();

  for (int i = 0; i < 3; i++) {

    relayOn();
    delay(10000);
    relayOff();

    delay(10000);

    float after = readMoisturePercent();

    if (after > before) return;
  }

  lockdown = true;
}

// =======================
// SETUP
// =======================

void setup() {
  Serial.begin(115200);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(200);

  display.begin(SSD1306_SWITCHCAPVCC, 0x3C);
  bmp.begin(0x77);

  pinMode(RELAY_PIN, OUTPUT);
}

// =======================
// LOOP (FIX)
// =======================

void loop() {
  unsigned long now = millis();

  static unsigned long lastCheck = 0;

  if (now - lastCheck > 5000) {
    lastCheck = now;

    refreshSensors();

    if (isWatering) {
      updateDisplay("LAISTYMAS", lastMoisture, lastTemp, lastPressure);
    }
    else if (lockdown) {
      updateDisplay("LOCK", lastMoisture, lastTemp, lastPressure);
    }
    else if (lastMoisture < 30.0) {
      autoWaterCycle();
    }
    else {
      updateDisplay("OK", lastMoisture, lastTemp, lastPressure);
    }
  }

  if (now - lastStatusSend > STATUS_INTERVAL) {
    sendStatus();
    lastStatusSend = now;
  }

  if (now - lastCmdCheck > 15000) {
    checkCommands();
    lastCmdCheck = now;
  }

  if (now - lastDataSend > 60000) {
    sendDataUsage();
    lastDataSend = now;
  }
}
